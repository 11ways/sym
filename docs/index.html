sym

A user-friendly symbolic link manager for ~/.local/bin

Installation

# Install using Homebrew (recommended) $ brew install 11ways/sym/sym # Verify installation $ sym --version sym version 1.0.1 # View documentation $ man sym
# One-line installation $ curl -fsSL https://raw.githubusercontent.com/11ways/sym/main/install.sh | bash # Or safer (inspect first) $ curl -fsSL https://raw.githubusercontent.com/11ways/sym/main/install.sh -o install.sh $ less install.sh $ bash install.sh # Or use Homebrew on Linux $ brew install 11ways/sym/sym
# Clone and install with Makefile $ git clone https://github.com/11ways/sym.git $ cd sym $ make install-local # Or quick setup $ chmod +x sym.sh $ ./sym.sh sym $(pwd)/sym.sh
